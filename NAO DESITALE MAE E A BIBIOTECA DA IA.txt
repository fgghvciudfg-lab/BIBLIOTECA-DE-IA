/*
  Biblioteca √∫nica:
  - Busca RSS via proxy "allorigins" para contornar CORS quando necess√°rio.
  - Filtra links locais com a caixa de busca.
  - Abre Google com termo pesquisado (abre nova aba).

// Lista de feeds (RSS) p√∫blicos ‚Äî adicione/remova conforme quiser
const RSS_FEEDS = [
  {id:"bbc-feed", url:"https://feeds.bbci.co.uk/news/rss.xml", title:"BBC"},
  {id:"cnn-feed", url:"http://rss.cnn.com/rss/cnn_topstories.rss", title:"CNN"},
  {id:"reuters-feed", url:"https://www.reutersagency.com/feed/?best-topics=top-news", title:"Reuters"},
  {id:"nasa-feed", url:"https://www.nasa.gov/rss/dyn/breaking_news.rss", title:"NASA"}
];

// Proxy para evitar CORS (AllOrigins). Se travar, pode trocar por https://api.allorigins.win/raw?url=
const PROXY = "https://api.allorigins.win/raw?url=";

// Fun√ß√£o para carregar cada feed e renderizar
async function loadFeed(feed){
  const target = document.getElementById(feed.id);
  if(!target) return;
  try{
    const resp = await fetch(PROXY + encodeURIComponent(feed.url));
    if(!resp.ok) throw new Error("fetch fail");
    const txt = await resp.text();
    const parser = new DOMParser();
    const xml = parser.parseFromString(txt, "text/xml");
    const items = Array.from(xml.querySelectorAll("item")).slice(0,8);
    if(items.length === 0){
      target.innerHTML = "<p class='small'>Sem itens (ou feed bloqueado)</p>";
      return;
    }
    const ul = document.createElement("ul");
    items.forEach(it=>{
      const title = it.querySelector("title") ? it.querySelector("title").textContent : "(sem t√≠tulo)";
      const link = it.querySelector("link") ? it.querySelector("link").textContent : null;
      const li = document.createElement("li");
      if(link){
        li.innerHTML = `<a href="${link}" target="_blank">${escapeHtml(title)}</a>`;
      } else {
        li.textContent = title;
      }
      ul.appendChild(li);
    });
    target.innerHTML = "";
    target.appendChild(ul);
  }catch(e){
    target.innerHTML = "<p class='small'>Erro ao carregar feed ‚Äî verifique CORS/proxy</p>";
    console.error("Feed error", feed.url, e);
  }
}

// carregamento inicial de RSS
RSS_FEEDS.forEach(feed => loadFeed(feed));

// BUSCA LOCAL (filtra todos os links do documento)
function filterLinks(){
  const q = document.getElementById("globalSearch").value.trim().toLowerCase();
  if(!q){
    // mostra tudo
    Array.from(document.querySelectorAll("main a")).forEach(a=>a.style.display="");
    return;
  }
  Array.from(document.querySelectorAll("main a")).forEach(a=>{
    const txt = (a.textContent + " " + (a.href || "")).toLowerCase();
    a.style.display = txt.includes(q) ? "" : "none";
  });
}

// pesquisa no Google (abre aba nova)
function openGoogle(){
  const q = document.getElementById("globalSearch").value.trim();
  const url = "https://www.google.com/search?q=" + encodeURIComponent(q || "");
  window.open(url, "_blank");
}

// util
function escapeHtml(s){ return s.replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;"); }

// opcional: monta lista de tags/categories lateral (extra)
(function buildTags(){
  const cats = Array.from(document.querySelectorAll("main h2")).map(h=>h.textContent.trim());
  const container = document.getElementById("tags");
  cats.forEach(c=>{
    const span = document.createElement("span");
    span.className = "tag";
    span.textContent = c;
    span.onclick = ()=>{ document.getElementById("globalSearch").value = c; filterLinks(); };
    container.appendChild(span);
  });
})();
